	hidden = yes
	immediate = {
		# Remove obsolete global state left by older MDE versions.
		remove_global_variable = no_dany_eggs

		if = {
			limit = {
				illyrio_has_eggs_for_exiles = yes
				NOT = {
					has_game_rule = agot_faegon_identity_open_blackfyre
				}
			}
			character:Mopatis_1 = {
				save_scope_as = egg_granter
				random_character_artifact = {
					limit = { has_variable = dragon_egg }
					save_scope_as = dany_transfer_egg_1
					add_to_list = dany_invasion_eggs
				}
				random_character_artifact = {
					limit = {
						has_variable = dragon_egg
						this != scope:dany_transfer_egg_1
					}
					save_scope_as = dany_transfer_egg_2
					add_to_list = dany_invasion_eggs
				}
				random_character_artifact = {
					limit = {
						has_variable = dragon_egg
						this != scope:dany_transfer_egg_1
						this != scope:dany_transfer_egg_2
					}
					add_to_list = dany_invasion_eggs
				}
			}
		}

			limit = {
				any_artifact = {
					has_variable = dragon_egg
					has_variable = dragon_embryo
					OR = {
						var:dragon_embryo = flag:drogon
						var:dragon_embryo = flag:viserion

					}

							has_variable = dragon_egg

						}
						any_character_artifact = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:viserion
						}
						any_character_artifact = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:rhaegal
						}
					}
				}
			}
			random_artifact = {
				limit = {
					has_variable = dragon_egg
					has_variable = dragon_embryo
					OR = {
						var:dragon_embryo = flag:drogon
						var:dragon_embryo = flag:viserion
						var:dragon_embryo = flag:rhaegal
					}
					this.artifact_owner = {
						any_character_artifact = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:drogon
						}
						any_character_artifact = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:viserion
						}
						any_character_artifact = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:rhaegal
						}
					}
				}
				artifact_owner = {
					save_scope_as = egg_granter
					random_character_artifact = {
						limit = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:drogon
						}
						add_to_list = dany_invasion_eggs
					}
					random_character_artifact = {
						limit = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:viserion
						}
						add_to_list = dany_invasion_eggs
					}
					random_character_artifact = {
						limit = {
							has_variable = dragon_egg
							has_variable = dragon_embryo
							var:dragon_embryo = flag:rhaegal

				faith = title:b_mummer_lane.faith
				culture = title:b_mummer_lane.culture

					min_template_decent_skill

					set_variable = agot_delete_char
				}
			}
		}
		trigger_event = agot_targaryen_exile_invasion.0301
	}

agot_targaryen_exile_invasion.0301 = {
	type = character_event
	title = agot_targaryen_exile_invasion.0301.t
	desc = agot_targaryen_exile_invasion.0301.desc
	override_icon = {
		reference = "gfx/interface/icons/event_types/type_war.dds"
	}
	theme = war
	left_portrait = {
		character = scope:egg_granter
		animation = toast_goblet
	}
	right_portrait = {
		character = root
		animation = disbelief
	}
	artifact = {
		target = scope:egg_1
		position = lower_left_portrait
	}
	artifact = {
		target = scope:egg_2
		position = lower_right_portrait
	}
	artifact = {
		target = scope:egg_3
		position = lower_center_portrait
	}
	immediate = {
		# Migrate a save made after the old MDE .0300 event.
		if = {
			limit = {
				list_size:dany_invasion_eggs < 3
				list_size:eggs_to_transfer > 0
			}
			if = {
				limit = {
					any_in_list = {
						list = eggs_to_transfer
						has_variable = dragon_embryo
						var:dragon_embryo = flag:drogon
					}
					any_in_list = {
						list = eggs_to_transfer
						has_variable = dragon_embryo
						var:dragon_embryo = flag:viserion
					}
					any_in_list = {
						list = eggs_to_transfer
						has_variable = dragon_embryo
						var:dragon_embryo = flag:rhaegal
					}
				}
				random_in_list = {
					list = eggs_to_transfer
					limit = {
						has_variable = dragon_embryo
						var:dragon_embryo = flag:drogon
					}
					add_to_list = dany_invasion_eggs
				}
				random_in_list = {
					list = eggs_to_transfer
					limit = {
						has_variable = dragon_embryo
						var:dragon_embryo = flag:viserion
					}
					add_to_list = dany_invasion_eggs
				}
				random_in_list = {
					list = eggs_to_transfer
					limit = {
						has_variable = dragon_embryo
						var:dragon_embryo = flag:rhaegal
					}
					add_to_list = dany_invasion_eggs
				}
			}
			else = {
				random_in_list = {
					list = eggs_to_transfer
					save_scope_as = dany_legacy_egg_1
					add_to_list = dany_invasion_eggs
				}
				random_in_list = {
					list = eggs_to_transfer
					limit = { this != scope:dany_legacy_egg_1 }
					save_scope_as = dany_legacy_egg_2
					add_to_list = dany_invasion_eggs
				}
				random_in_list = {
					list = eggs_to_transfer
					limit = {
						this != scope:dany_legacy_egg_1
						this != scope:dany_legacy_egg_2
					}
					add_to_list = dany_invasion_eggs
				}
			}
		}

		if = {
			limit = {
				list_size:dany_invasion_eggs > 2
			}
			every_in_list = {
				list = dany_invasion_eggs
				set_owner = root
				set_variable = {
					name = destined_eggs
					value = yes
				}
				set_variable = {
					name = agot_artifact_cannot_gift
					value = yes
				}
				set_variable = {
					name = dany_invasion_selected_egg
					value = yes
				}
				generated_eggs_cleanup_effect = yes
				add_to_list = generated_eggs
			}
		}
		else = {
			# Mark pre-existing eggs so only the three eggs created below
			# can enter the ceremony.
			every_character_artifact = {
				limit = { has_variable = dragon_egg }
				set_variable = {
					name = dany_invasion_preexisting_egg
					value = yes
				}
			}
			agot_spawn_laid_egg = {
				OWNER_SCOPE = root
				AMOUNT_OF_EGGS_TO_SPAWN = 3
				DRAGON_PARENT = character:dragon_pythax
			}
			every_character_artifact = {
				limit = {
					has_variable = dragon_egg
					NOT = {
						has_variable = dany_invasion_preexisting_egg
					}
				}
				set_variable = {
					name = destined_eggs
					value = yes
				}
				set_variable = {
					name = agot_artifact_cannot_gift
					value = yes
				}
				set_variable = {
					name = dany_invasion_selected_egg
					value = yes
				}
				set_variable = {
					name = dany_invasion_fallback_egg
					value = yes
				}
				generated_eggs_cleanup_effect = yes
				add_to_list = generated_eggs
			}
			every_character_artifact = {
				limit = {
					has_variable = dany_invasion_preexisting_egg
				}
				remove_variable = dany_invasion_preexisting_egg
			}
			custom_tooltip = agot_targaryen_exile_invasion.0301.generated_eggs
		}

		every_in_list = {
			list = generated_eggs
			limit = {
				has_variable = dany_invasion_selected_egg
			}
			if = {
				limit = {
					NOT = { exists = scope:egg_1 }
				}
				save_scope_as = egg_1
			}
			else_if = {
				limit = {
					NOT = { exists = scope:egg_2 }
				}
				save_scope_as = egg_2
			}
			else_if = {
				limit = {
					NOT = { exists = scope:egg_3 }
				}
				save_scope_as = egg_3
			}
		}

		if = {
			limit = {
				player_heir.house ?= root.house
			}
			player_heir = { save_scope_as = heir }
		}
	}

	# A wonderful gift
	option = {
		name = agot_targaryen_exile_invasion.0301.a
		custom_tooltip = agot_targaryen_exile_invasion.0301.teaser_eggs
		ai_chance = {
			base = 75
			ai_value_modifier = {
				ai_boldness = 2
				ai_greed = -2
			}
		}
	}


	after = {
		trigger_event = { id = agot_targaryen_exile_invasion.0600 days = 7 }
		hidden_effect = {
			if = {
				limit = {
					scope:egg_granter = {
						has_variable = agot_delete_char
					}
				}
				scope:egg_granter = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}
}
