dragon_age_svalue = {
	value = {
		if = {
			limit = {
				has_character_flag = currently_riding_dragon
				agot_has_dragon_storage_system_global_list = yes
			}
			every_in_global_list = {
				variable = gl_dragon_variable_storage
				limit = {
					var:dragon_id ?= root.var:current_dragon
				}
				save_temporary_scope_as = dragon_var_story_val
			}
			value = scope:dragon_var_story_val.var:dragon_age
		}
		if = {
			limit = {
				agot_has_dragon_storage_system_global_list = yes
			}
			every_in_global_list = {
				variable = gl_dragon_variable_storage
				limit = {
					var:dragon_id ?= root
				}
				save_temporary_scope_as = dragon_var_story_val
			}
			value = scope:dragon_var_story_val.var:dragon_age
		}
	}
}

gene_dragon_primary_color_value_svalue = {
	value = {
		# AGOT DRAGON GENE SAFETY SAFE 0.5.2
		# Fallback visual: solo se usa si no existe un registro o variable válida.
		value = 0

		# Retrato del jinete: consulta únicamente un current_dragon válido.
		if = {
			limit = {
				has_character_flag = currently_riding_dragon
				has_variable = current_dragon
				agot_has_dragon_storage_system_global_list = yes
			}
			every_in_global_list = {
				variable = gl_dragon_variable_storage
				limit = {
					var:dragon_id ?= root.var:current_dragon
					has_variable = gene_dragon_primary_color_value # AGOT LOCAL MDE DUPLICATE FILTER 0.5.4
				}
				save_temporary_scope_as = dragon_var_story_val
			}
			if = {
				limit = {
					exists = scope:dragon_var_story_val
					scope:dragon_var_story_val = {
						has_variable = gene_dragon_primary_color_value
					}
				}
				value = scope:dragon_var_story_val.var:gene_dragon_primary_color_value
			}
		}

		# Retrato del dragón: evita ejecutar esta rama sobre personajes humanos.
		if = {
			limit = {
				has_trait = dragon
				agot_has_dragon_storage_system_global_list = yes
			}
			every_in_global_list = {
				variable = gl_dragon_variable_storage
				limit = {
					var:dragon_id ?= root
					has_variable = gene_dragon_primary_color_value # AGOT LOCAL MDE DUPLICATE FILTER 0.5.4
				}
				save_temporary_scope_as = dragon_var_story_val
			}
			if = {
				limit = {
					exists = scope:dragon_var_story_val

						has_variable = gene_dragon_primary_color_value
