agot_is_valid_vhagar_birth = {
	$EGG$ = { save_temporary_scope_as = temp_egg }
	NOT = { exists = scope:vhagar_birth_reserved }
	current_date >= 7949.1.1
	scope:temp_egg = {
		var:dragon_parent ?= { is_character_dragon_nessarion = yes }
	}
	NOT = { exists = character:dragon_vhagar }
	NOT = {
		any_artifact = { # MDE Modified
			has_variable = dragon_embryo
			this.var:dragon_embryo = flag:vhagar
		}
	}
	NOT = {
		any_in_global_list = {
			variable = gl_dragon_variable_storage
			var:dragon_id ?= { has_inactive_trait = is_dragon_vhagar }
		}
	}
}
