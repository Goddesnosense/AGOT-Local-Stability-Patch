# AGOT Local Stability Patch 0.5.6 - canon children compatibility

agot_canon_children_start_pregnancy_basic_trigger = {
	$MOTHER$ = { save_temporary_scope_as = mother }
	$FATHER$ = { save_temporary_scope_as = father }

	scope:mother = {
		OR = {
			is_spouse_of = scope:father
			has_relation_lover = scope:father
			is_concubine_of = scope:father # MDE Modified
		}
	}
}

agot_canon_children_child_pregnancy_trigger = {
	save_temporary_scope_as = mother
	NOT = { has_character_flag = agot_story_cycle_blocking_canon_children }
	NOT = {
		agot_canon_children_had_child_trigger = {
			ID = $ID$
			MOTHER = scope:mother
		}
	}
	agot_canon_children_birth_year_trigger = {
		BIRTH_YEAR = $BIRTH_YEAR$
		BIRTH_YEAR_MIN = $BIRTH_YEAR_MIN$
	}
	is_adult = yes
	age >= 16
	is_alive = yes
	is_pregnant = no
	can_have_children = yes
	is_imprisoned = no

	scope:canon_father = {
		is_adult = yes
		age >= 16
		can_have_children = yes
		is_imprisoned = no
		# No booty calls at the wall
		NOT = { has_trait = nightswatch }
	}
}
