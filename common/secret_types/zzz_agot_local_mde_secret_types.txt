# AGOT Local Stability Patch 0.5.6 - Hightower hidden artifact hook

secret_agot_hidden_artifact = {
	category = artifact

	is_valid = {
		secret_agot_hidden_artifact_is_valid_trigger = {
			OWNER = scope:secret_owner
		}
	}

	is_shunned = {
		secret_agot_hidden_artifact_is_shunned_trigger = {
			OWNER = scope:secret_owner
		}
	}

	is_criminal = {
		secret_agot_hidden_artifact_is_criminal_trigger = {
			OWNER = scope:secret_owner
		}
	}

	on_expose = {
		save_scope_as = secret
		scope:secret_owner = {
			if = {
				limit = { house ?= dynasty:dynn_Uller.dynasty_founder.house }
				trigger_event = agot_secret_events.3000
			}
			else_if = {
				limit = {
					exists = dynasty:dynn_Mopatis
					house ?= dynasty:dynn_Mopatis.dynasty_founder.house
				}
				trigger_event = agot_secret_events.3010
			}
			else_if = {
				limit = { house ?= dynasty:dynn_Hightower.dynasty_founder.house }
				trigger_event = mde_decisions_events.3000
			}
		}
		secret_exposed_notification_effect = yes
	}
}
