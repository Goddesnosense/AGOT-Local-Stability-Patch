agot_hf_secret_thing = {
	save_scope_as = the_ruler
	if = {
		limit = {
			scope:the_ruler = {
				is_human = yes
				NOT = { government_has_flag = government_is_uninteractable }
				exists = house
				NOT = { has_character_flag = agot_hf_secret_thing_already_happened }
				is_landed = yes
				agot_can_be_ruler = yes
				highest_held_title_tier > tier_county
				any_sibling = {
					count >= 1
					AND = {
						exists = house
						is_alive = yes
						agot_hf_special_secret_character_eligible = yes
						NOT = { has_character_flag = agot_is_at_unknown_place }
						any_child = { count = 0 }
						age >= 10
						age < 40
						agot_can_be_ruler = yes
						is_playable_character = no
						AND = {
							save_temporary_opinion_value_as = {
								name = opinion_on_the_ruler
								target = scope:the_ruler
							}
							scope:opinion_on_the_ruler <= 30
						}
						NOT = { is_married = yes }
						NOT = {
							any_child = {
								count > 0
							}
						}
						NOT = { this = character:Targaryen_27 }
						NOT = { this = character:Targaryen_28 }
						NOT = { this = character:Targaryen_29 }
					}
				}
			}
		}
		add_character_flag = agot_hf_secret_thing_already_happened
		random_sibling = {
			limit = {
				AND = {
					exists = house
					is_alive = yes
					agot_hf_special_secret_character_eligible = yes
					NOT = { has_character_flag = agot_is_at_unknown_place }
					any_child = { count = 0 }
					age >= 10
					age < 40
					agot_can_be_ruler = yes
					is_playable_character = no
					AND = {
						save_temporary_opinion_value_as = {
							name = opinion_on_the_ruler
							target = scope:the_ruler
						}
						scope:opinion_on_the_ruler <= 30
					}
					NOT = { is_married = yes }
					NOT = {
						any_child = {
							count > 0
						}
					}
					NOT = { this = character:Targaryen_27 }
					NOT = { this = character:Targaryen_28 }
					NOT = { this = character:Targaryen_29 }
				}
			}
			save_scope_as = the_sibling
			random = {
				chance = {
					value = 0
					add = {
						value = scope:the_sibling.ai_energy
						divide = 2
						min = 1
					}
				}
				trigger_event = { id = agot_hf_generic_events.0001 days = 40 }
			}
		}
	}
}

agot_hf_revealbastard_story_end_effect = {
	# AGOT_LOCAL_STABILITY_HF_BASTARD_CADET_SAFE_010_BEGIN
	if = {
		limit = {
			exists = scope:bastard_character.top_liege.primary_title
		}

		hidden_effect = {
			if = {
				limit = { scope:bastard_character.top_liege.primary_title ?= { has_variable = agot_hf_revealbastard_started } }
				scope:bastard_character.top_liege.primary_title = { remove_variable = agot_hf_revealbastard_started }
			}
			if = {
				limit = { scope:bastard_character.top_liege.primary_title ?= { has_variable = agot_hf_revealbastard_started_ai } }
				scope:bastard_character.top_liege.primary_title = {
					remove_variable = agot_hf_revealbastard_started_ai
					set_variable = {
						name = agot_hf_revealbastard_started_ai
						years = 1
					}
				}
			}
			scope:story = { end_story = yes }
		}

	}
	# AGOT_LOCAL_STABILITY_HF_BASTARD_CADET_SAFE_010_END
}

agot_hf_revealbastard_liege_finder = {

		}

			}

	}

}

			#IS A BASTARD!

						NOT = { exists = scope:bastard_character.father }

				}

				set_father = scope:bastard_character.real_father
				set_parent_house_effect = yes

					limit = {
						has_trait = disputed_heritage
						is_dynast = yes

							scope:bastard_character.house = scope:bastard_character.mother.house

						}

						if = {
							limit = {
								has_trait = disputed_heritage
								exists = scope:bastard_character.real_father.house
								scope:bastard_character.real_father.dynasty = {
									any_dynasty_member = {
										count >= 1
									}
								}
							}
							set_house = real_father.house
							save_scope_as = father_first
						}

							limit = {
								has_trait = disputed_heritage
								OR = {

								}

agot_hf_cadet_generation = {
	if = {
		limit = {
			OR = {
				exists = mother.dynasty
				exists = father.dynasty
			}

			trigger_if = {
				limit = {
					exists = father
					exists = father.house
					NOT = { father = { has_trait = bastard } }
				}
				father.house = { exists = house_head }
			}
			trigger_else_if = {
				limit = {
					exists = mother
					exists = mother.house
					NOT = { mother = { has_trait = bastard } }
				}
				mother.house = { exists = house_head }
			}
			trigger_else = { always = no }
		}

		house = { save_scope_as = generated_house }
		if = {
			limit = {
				is_lowborn = no
				exists = house
			}
			house = { save_scope_as = former_house }
		}

			if = {
				limit = {
					scope:former_house = { has_house_modifier = dragonrider_house_modifier }
				}
				add_house_modifier = dragonrider_house_modifier
			}
		}

	}
